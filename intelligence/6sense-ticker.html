Python Data Pipeline Code

# Python script to fetch 6sense data via API or scraping import requests import json from selenium import webdriver from selenium.webdriver.common.by import By from selenium.webdriver.support.ui import WebDriverWait import time import pandas as pd class SixSenseScraper: def __init__(self, segment_id, session_cookies): self.segment_id = segment_id self.base_url = f"https://kofax.abm.6sense.com/segments/segment/{segment_id}" self.cookies = session_cookies self.accounts = [] self.keywords = {} self.analytics = {} def scrape_accounts(self, max_pages=10): driver = webdriver.Chrome() driver.get(f"{self.base_url}/accounts") # Add cookies for auth for cookie in self.cookies: driver.add_cookie(cookie) all_accounts = [] for page in range(1, max_pages + 1): time.sleep(3) rows = driver.find_elements(By.CSS_SELECTOR, "table.account-table tr") for row in rows[1:]: # Skip header cols = row.find_elements(By.TAG_NAME, "td") account_data = { "name": cols[0].text, "location": cols[1].text, "buying_stage": cols[2].text, "visitors": cols[3].text, "keywords": cols[4].text, "employees": cols[5].text, "revenue": cols[6].text, "industry": cols[7].text } all_accounts.append(account_data) # Click next page try: next_btn = driver.find_element(By.CSS_SELECTOR, ".pagination .next") next_btn.click() except: break driver.quit() self.accounts = all_accounts return all_accounts def scrape_analytics(self): driver = webdriver.Chrome() driver.get(f"{self.base_url}/analytics") time.sleep(5) # Scrape Behavioral Analytics keywords driver.find_element(By.LINK_TEXT, "Behavioral Analytics").click() time.sleep(3) keyword_elements = driver.find_elements(By.CSS_SELECTOR, ".keyword-item") keywords = [kw.text for kw in keyword_elements] self.analytics["keywords"] = keywords driver.quit() return self.analytics def export_to_json(self, filename): data = { "segment_id": self.segment_id, "accounts": self.accounts, "analytics": self.analytics } with open(filename, "w") as f: json.dump(data, f, indent=2) def export_to_csv(self, filename): df = pd.DataFrame(self.accounts) df.to_csv(filename, index=False) # Usage scraper = SixSenseScraper(segment_id="2430296", session_cookies=YOUR_COOKIES) scraper.scrape_accounts(max_pages=9) scraper.scrape_analytics() scraper.export_to_json("6sense_data.json") scraper.export_to_csv("6sense_accounts.csv")
